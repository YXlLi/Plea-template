\begin{frame}\frametitle{2.1\ 圆形场的HHG中的自旋守恒 }
\begin{center}Model 1\end{center}
\begin{equation}
\textbf{E}=\frac{E_{0}e^{i\omega t}}{2\sqrt{2}}(\frac{1+\epsilon}{\sqrt{1+\epsilon^{2}}}\hat{\textbf{e}}_{R}+\frac{1-\epsilon}{\sqrt{1+\epsilon^{2}}}\hat{\textbf{e}}_{L})\label{field}
\end{equation}
\begin{equation}
\Omega_{(n_{1},n_{2})}=n_{1}\omega+n_{2}r\omega
\end{equation}
\begin{equation}
\langle\hat{\sigma}\rangle=\frac{2\epsilon}{1+\epsilon^2}\footnote{$\langle\hat{\sigma}\rangle=\sin{(2\alpha)}$,$\epsilon=\tan{(\alpha)}$}
\end{equation}
\begin{equation}
\langle\sigma_{(n_{1},n_{2})}\rangle=n_{1}\langle\sigma_{1}\rangle+n_{2}\langle\sigma_{2}\rangle+\delta_{(n_{1},n_{2})}
\end{equation}
\end{frame}

\begin{frame}\frametitle{2.1\ 圆形场的HHG中的自旋守恒 }
\begin{center}Model 2\end{center}
\begin{equation}
\textbf{E}=\frac{E_{0}e^{i\omega t}}{2}(\cos{(\delta\alpha)}\hat{\textbf{e}}_{R}+\sin{(\delta\alpha)}\hat{\textbf{e}}_{L})+c.c\footnote{$\delta\alpha=\alpha-\pi/4$,$\epsilon=\tan{(\alpha)}$}
\end{equation}
\begin{equation}
\Omega_{(n_{+},n_{-};n_{2})}=(n_{+}+n_{-})\omega+n_{2}r\omega
\end{equation}
\begin{equation}
\sigma_{(n_{+},n_{-};n_{2})}=n_{+}\sigma_{+}+n_{-}\sigma_{-}+n_{2}\sigma_{2}\footnote{$\sigma_{+}=+1$，$\sigma_{-}=\sigma_{2}=-1$}
\end{equation}

\end{frame}
